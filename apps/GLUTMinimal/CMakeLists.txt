cmake_minimum_required(VERSION 2.6)

project( GLUTMinimal )

find_package( OpenGL REQUIRED )
find_package( GLUT REQUIRED )
find_package( GLEW REQUIRED )

find_package( Boost COMPONENTS program_options REQUIRED )

set( SOURCES
  src/main.cpp
)

set ( HEADERS
)

include_directories( "."
                     "inc"
                     ${Boost_INCLUDE_DIRS}
                     ${GLEW_INCLUDE_DIRS}
                     ${GLUT_INCLUDE_DIR}
                   )

add_definitions( ${GLEW_DEFINITIONS} )

add_executable( GLUTMinimal
  ${SOURCES}
  ${HEADERS}
)

target_link_libraries( GLUTMinimal
  ${GLUT_LIBRARIES}
  ${OPENGL_gl_LIBRARY}
  ${GLEW_LIBRARY}
  ${Boost_LIBRARIES}
  DPSgCore
  DPSgGL
  DPSgRdrRiXGL
  DPSgIO
  DPSgUIManipulator
  DPSgGenerator
  DPSgUIGLUT
)

CopyGLEW( GLUTMinimal "${DP_BINARY_PATH}" )
CopyGLUT( GLUTMinimal "${DP_BINARY_PATH}" )

set_target_properties( GLUTMinimal PROPERTIES FOLDER "Apps")
set_target_properties( GLUTMinimal PROPERTIES LINK_FLAGS "/NODEFAULTLIB:\"freeglutd.lib\" /NODEFAULTLIB:\"freeglut.lib\"" )
